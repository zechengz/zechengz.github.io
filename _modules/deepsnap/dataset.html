    def __init__(self, graphs: List[Graph], task: str='node',
            edge_negative_sampling_ratio: float=1,
            edge_message_ratio: float=0.2,
            edge_train_mode: str='all'):
        r"""
        A plain python object modeling a list of Graph with various
        (optional) attributes.

        Args:
            graphs: a list of Graph
            task: task this GraphDataset is used for 
                (task = 'node' or 'edge' or 'link_pred' or 'graph')
            edge_negative_sampling_ratio: number of negative samples compared
                to that of positive data
            edge_message_ratio: number of training edge objectives 
                compared to that of message-passing edges
            edge_train_mode: whether to use 
                (edge_train_mode = 'all': 
                    training edge objectives are the same as the message-passing 
                    edges
                or 'disjoint': training edge objectives are different from 
                    message-passing edges
                or 'train_only': training edge objectives are always the 
                    training set edges)
        """
        self.graphs = graphs
        if graphs is None:
            raise ValueError('Graphs are None')
        self.task = task
        self.edge_message_ratio = edge_message_ratio
        self.edge_negative_sampling_ratio = edge_negative_sampling_ratio
        self.edge_train_mode = edge_train_mode
        self._num_node_labels = None
        self._num_edge_labels = None
        self._resample_negatives = False

    def __len__(self) -> int:
        r"""
        Returns:       
            the number of graph in graphs.
        """
        return len(self.graphs)

    @property
    def num_node_features(self) -> int:
        r"""
        Returns:
            the number of features per node in the dataset.
        """
        return self.graphs[0].num_node_features

    @property
    def num_edge_features(self) -> int:
        r"""
        Returns:
            the number of features per edge in the dataset.
        """
        return self[0].num_edge_features

    @property
    def num_node_labels(self) -> int:
        r"""
        Returns:
            the number of labels per node in the dataset.
        """
        if self._num_node_labels is None:
            self._num_node_labels = \
                max([graph.num_node_labels for graph in self.graphs])
        return self._num_node_labels

    @property
    def num_edge_labels(self) -> int:
        r"""
        Returns:
            the number of labels per node in the dataset.
        """
        if self._num_edge_labels is None:
            self._num_edge_labels = \
                max([graph.num_edge_labels for graph in self.graphs])
        return self._num_edge_labels

[docs]    def split(self, transductive: bool=True, split_ratio: List[float]=[0.8, 0.1, 0.1]) -> List[Graph]:
        r"""
        Args:
            transductive: whether the training process is transductive
                or inductive
            split_ratio: number of data splitted into train, validation 
                (and test) set
        Returns:
            a list of 3 (2) lists of graph object corresponding to train, 
                validation (and test) set
        """

        if transductive:
            if self.task == 'graph':
                raise ValueError('Graph prediction task cannot be transductive')
            # a list of split graphs
            # (e.g. [[train graph, val graph, test graph], ... ])
            graphs_split_list = []
            for graph in self.graphs:
                graphs_split = graph.split(self.task, split_ratio)
                if self.task == 'link_pred' and \
                        self.edge_train_mode == 'disjoint':
                    graphs_split[0] = graphs_split[0]._split_link_pred(
                            self.edge_message_ratio)[1]
                graphs_split_list.append(graphs_split)

            # list of num_splits datasets
            # (e.g. [train dataset, val dataset, test dataset])
            dataset_return = []
            for x in zip(*graphs_split_list):
                dataset_current = copy.copy(self)
                dataset_current.graphs = list(x)
                if self.task == 'link_pred':
                    for graph_temp in dataset_current.graphs:
                        graph_temp._create_label_link_pred(
                                self.edge_negative_sampling_ratio)
                dataset_return.append(dataset_current)
        else:
            # inductive split
            self._shuffle()
            # a list of num_splits list of graphs
            # (e.g. [train graphs, val graphs, test graphs])
            split_graphs = []
            split_offset = 0
            num_graphs = len(self.graphs)
            for i in range(len(split_ratio)):
                if i != len(split_ratio) - 1:
                    num_split_i = int(split_ratio[i] * num_graphs)
                    split_graphs.append(
                        self.graphs[split_offset: split_offset + num_split_i])
                    split_offset += num_split_i
                else:
                    split_graphs.append(self.graphs[split_offset:])

            # create objectives for link_pred task
            if self.task == 'link_pred':
                # if disjoint, this will split each all graph's edges into 2:
                # message passing and objective edges
                # which is returned by the [1] of the split graphs
                if self.edge_train_mode == 'disjoint':
                    for i in range(len(split_graphs)):
                        for j in range(len(split_graphs[i])):
                            split_graphs[i][j] = \
                                split_graphs[i][j]._split_link_pred(
                                    self.edge_message_ratio)[1]
                # if all, train graphs will not be split
                elif self.edge_train_mode == 'all':
                    for i in range(1,len(split_graphs)):
                        for j in range(len(split_graphs[i])):
                            split_graphs[i][j] = \
                                split_graphs[i][j]._split_link_pred(
                                    self.edge_message_ratio)[1]

            # list of num_splits datasets
            dataset_return = []
            for graphs in split_graphs:
                dataset_current = copy.copy(self)
                dataset_current.graphs = graphs
                if self.task == 'link_pred':
                    for graph_temp in dataset_current.graphs:
                        graph_temp._create_label_link_pred(
                            self.edge_negative_sampling_ratio)
                dataset_return.append(dataset_current)
        # resample negatives for train split (only for link prediction)
        dataset_return[0]._resample_negatives = True

        return dataset_return

            transform: user-defined transformation function

        """
        # TODO: parallel apply
        for graph in self.graphs:
            graph.apply_transform(transform, graph_fixed, **kwargs)



[docs]    def to(self, device):
        r"""
        transfer Graph object in the graphs to specified device
        """
        for graph in self.graphs:
            graph.to(device)


    def _shuffle(self):
        r"""
        shuffle Graph object in graphs 
        """
        random.shuffle(self.graphs)


[docs]    @staticmethod
    def pyg_to_graphs(dataset) -> List[Graph]:
        r"""
        Transform a list of torch_geometric.data.Dataset object
            to a list of Graph object

        Args:
            a list of torch_geometric.data.Dataset object
        Returns:
            a list of Graph object
        """
        return [Graph.pyg_to_graph(data) for data in dataset]

    def list_to_graphs(G_list) -> List[Graph]:
        r"""

            to a list of Graph object

        Args:
            a list of networkx data object

    def __getitem__(self, idx) -> Union[Graph, List[Graph]]:
        r"""
        Takes in an integer (or a list of integers)
        returns a single Graph object (a subset of graphs)

        Args:
            idx: index to be selected from graphs

        Returns:
            a single Graph object or subset of graphs
        """

        if self.task == 'link_pred' and self._resample_negatives:
            # resample negative examples
            for graph in self.graphs:
                graph._create_label_link_pred(
                        self.edge_negative_sampling_ratio, resample=True)
        if isinstance(idx, int):
            return self.graphs[idx]
        else:
            return self._index_select(idx)

    def _index_select(self, idx) -> List[Graph]:
        r"""
        Takes in a list of integers, returns a subset of graphs
        corresponding to the list of integers

        Args:
            idx: index to be selected from graphs

        Returns:
            a single Graph object or subset of graphs
        """

        if isinstance(idx, slice):
            dataset = copy.copy(self)
            dataset.graphs = self.graphs[idx]
            return dataset
        elif torch.is_tensor(idx):
            if idx.dtype == torch.long:
                return self._index_select(idx.tolist())
            elif idx.dtype == torch.bool or idx.dtype == torch.uint8:
                return self._index_select(idx.nonzero().flatten().tolist())
        elif isinstance(idx, list) or isinstance(idx, tuple):
            dataset = copy.copy(self)
            dataset.graphs = [self.graphs[i] for i in idx]
            return dataset
        else:
            raise IndexError(
                    'Only integers, slices (`:`), list, tuples, and long or bool '
                    'tensors are valid indices (got {}).'.format(
                    type(idx).__name__))

    def __repr__(self) -> str:  # pragma: no cover
        return f'{self.__class__.__name__}({len(self)})'
